    Usage with handler # 

    type Props struct {}

func ExamplePage() Node {
	return Div(Text("Example page"))
}

func ExampleHandler() hiro.Handler {
	return func(c hiro.Ctx) error {
	    return c.Response().Render(ExamplePage(c, Props{}))	
    } 
}
